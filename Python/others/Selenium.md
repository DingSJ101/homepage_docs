sudo apt-get install firefox
pip install selenium
# install geckodriver 
## reference: https://firefox-source-docs.mozilla.org/testing/geckodriver/Support.html

tar -xvzf geckodriver-v0.31.0-linux32.tar.gz
sudo ln -s ~/workspace/geckodriver /usr/bin
